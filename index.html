import pandas as pd
import numpy as np

# Load model-ready revenue series
rev = pd.Series([2.12, 2.23, 2.31, 2.48], index=pd.period_range('2024Q3', periods=4, freq='Q'))

def yoy_growth(series):
    return (series.iloc[-1] / series.iloc[-5]) - 1 if len(series) > 4 else (series.iloc[-1] / series.iloc[0]) - 1

# Q2 YoY Growth
print(f"YoY: {100*yoy_growth(rev):.1f}%")
